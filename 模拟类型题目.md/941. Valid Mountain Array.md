class Solution:
    def validMountainArray(self, arr: List[int]) -> bool:

        i = 0
        
        while i+1 < len(arr) and arr[i] < arr[i+1]:
            i = i+1
            
        if i == 0 or i == len(arr) - 1:
            return False
        
        
        while i+1 < len(arr) and arr[i] > arr[i+1]:
            i = i +1
        
        if i == len(arr)-1:
            return True
        else:
            return False
