这道题的递归不好想 只能一个条件一个条件去考虑

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        
        #consider base case always for a recursive solving
        if root == None:
            return None
        
        if p == root or q == root:
            return root
        
        left = self.lowestCommonAncestor(root.left,p,q)
        right = self.lowestCommonAncestor(root.right,p,q)
        
        if left == None and right == None:
            return None
        
        #都在右子树
        elif left == None and right != None:
            return right
        
        #都在左子树
        elif left != None and right == None:
            return left
        
        return root
