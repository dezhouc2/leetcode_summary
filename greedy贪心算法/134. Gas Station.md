time:o(n)
space:o(1)

思路：
“travel around the circuit once in the clockwise direction” -> via example1, we know that if sum(gas) < sum(cost) -> return -1


class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        
        if sum(gas) < sum(cost):
            return -1
        
        curr = 0
        start = 0
        for i in range(len(gas)):
            curr = curr + gas[i] - cost[i] ##current gas 
            if curr < 0: ## if current gas is negative we change to next
                curr = 0
                start = i +1  ##we start from i+1 not start+1
        
        return start
