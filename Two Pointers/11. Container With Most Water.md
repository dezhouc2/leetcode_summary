
```python
time:o(n)
space:o(1)


class Solution:
    def maxArea(self, height: List[int]) -> int:
        
        #maximum = longest length * longest width
        
        i,j = 0,len(height)-1
        result = 0
        
        while i < j:
            leng = j-i
            width = min(height[j], height[i])
            curr = leng*width
            
            if height[i] < height[j]:
                result = max(result,curr)
                i = i +1
            
            else:
                result = max(result,curr)
                j = j-1
        
        return result
```
