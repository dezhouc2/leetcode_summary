提升:
1.可以用tuples来代替dictionary

2.如果用一个for loop怎么确保可以找到i !=j使得增加他们的和？
因为只有两个 可以通过排序然后找 i,i-1即可

class Solution:
    def maximumSum(self, nums: List[int]) -> int:
        
        result_ = [(sum(map(int,str(x))),x) for x in nums]
        result_.sort()
        
        #print(result_)
        
        count = -1
        for i in range(1,len(result_)):
            if result_[i][0] == result_[i-1][0]:
                count = max(count, result_[i][1]+result_[i-1][1])
        
        
        return count
