class Solution:
    def largestGoodInteger(self, num: str) -> str:
        
        val = "000"
        check = False
        
        for ele in num:
            if ele*3 in num:
                check = True
        
        if check != True:
            return ""
        
        for ele in num:
            if ele*3 in num:
                curr = ele*3
                val = max(val,curr)
        
        return val

time:o(n)
space:o(1)

总结:
1.看到string联想string的操作 -> 连续3个element是否在string里面 -> 用ele*3 in string 即可
2.max(a,b) 也可以用来对比 string type的整数
