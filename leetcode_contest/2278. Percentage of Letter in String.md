知识点:
math.floor(val) rounds a number down to the nearest integer(不管是否四舍五入, round(val)会自动四舍五入)

example:
val = 16.6
print(round(val))  #17
print(math.floor(val))  #16

class Solution:
    def percentageLetter(self, s: str, letter: str) -> int:
        
        count = s.count(letter)
        
        if letter not in s:
            return 0
        
        val = (count/len(s))*100
        return math.floor(val)
