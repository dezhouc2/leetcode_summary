
```python

知识点:
1. s.strip() 去掉string中前面和后面的空格
2. s.isalnum() 判断s是否为alphabetic 或 number
3. s.lower()

class Solution:
    def isPalindrome(self, s: str) -> bool:
        
        s = s.strip().lower()
        
        result =[]
        for c in s:
            if c.isalnum():
                result.append(c)
        
        result = "".join(result)
        
        if result == result[::-1]:
            return True
        else:
            return False
```
