“AB" 怎么得到28 -> 26*1 + 2
"A" 怎么得到 1 -> ord("A") - ord("A")+1

class Solution:
    def titleToNumber(self, columnTitle: str) -> int:
        
        #"AB" = 26*1 + 2
        count = 0
        result = 0
        
        for ele in columnTitle:
            count = 26*count + ord(ele) - ord("A")+1
        
        
        return count
