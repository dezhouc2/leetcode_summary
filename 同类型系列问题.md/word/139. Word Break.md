class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        
        @functools.cache
        
        def dfs(l,index):
            #递归结束的条件
            if index == len(s):
                return True
            
            #当前面的子字符串是符合wordDict的然后想要在这个基础上看看 剩下的字符串是否符合wordDict
            for i in range(index,len(s)):
                if s[index:i+1] in wordDict and dfs(s,i+1):
                    return True
            
            return False
            
        
        return dfs(s,0)
