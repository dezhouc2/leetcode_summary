class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        
        #we can use dictinoary
        countm = Counter(magazine)
        
        for ele in ransomNote:
            if ele in countm:
                if countm[ele] == 0:
                    return False
                countm[ele] = countm[ele]-1
            else:
                return False
        
        return True

思路:
这里用了counter, 当counter的某一个元素的频率为0的时候 并不会在counter里面减去这个元素的存在
理论上也可以用dictionary

自己的做法:
两个counter;一个counter去排除一下不在magazine的元素; 另外一个counter考虑元素是否可以形成ransomnote
time:o(n)
space:o(n+m)

class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        
        count_r = Counter(ransomNote)
        count_m = Counter(magazine)
        
        for ele in count_r:
            if ele not in count_m:
                return False
        
        for ele in count_m:
            if count_m[ele] < count_r[ele]:
                return False
        
        
        return True
