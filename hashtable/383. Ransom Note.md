class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        
        #we can use dictinoary
        countm = Counter(magazine)
        
        for ele in ransomNote:
            if ele in countm:
                if countm[ele] == 0:
                    return False
                countm[ele] = countm[ele]-1
            else:
                return False
        
        return True

思路:
这里用了counter, 当counter的某一个元素的频率为0的时候 并不会在counter里面减去这个元素的存在
理论上也可以用dictionary
