```python
time:o(logm+logn)
space: o(1)

思路:
从target考虑 -> 某一行的某一个列, 某一列中可以用binary search找到 -> 现在考虑某一行 
Integers in each row are sorted from left to right -> 可以联想到用binary search
The first integer of each row is greater than the last integer of the previous row -> 用来给binary search中某一个行提供判断


class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        
        row = len(matrix)
        col = len(matrix[0])
        
        L, R = 0, row-1
        currmid = 0
        
        while L <= R:
            Mid = (L+R)//2
            if target > matrix[Mid][-1]:
                L = Mid+1
            
            elif target < matrix[Mid][0]:
                R = Mid-1
            
            else:
                currmid = Mid
                break
        
        #edge case
        if L > R:
            return False
        
        #look for mid in a row
        l,r = 0, col -1
        
        while l <= r:
            mid = (l+r)//2
            if target > matrix[currmid][mid]:
                l = mid + 1
            elif target < matrix[currmid][mid]:
                r = mid -1
            
            else:
                return True
        
        return False
```
