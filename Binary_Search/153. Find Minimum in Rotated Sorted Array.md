思路:
考虑base case -> 当nums[r] > nums[l]的时候 因为nums是sorted 说明最小值为nums[0]

通过examples可以考虑到 如果nums[mid] > nums[l]的话说明要考虑的数值在mid后面

这里nums[mid]是用来 更新最后答案的最小值

class Solution:
    def findMin(self, nums: List[int]) -> int:
        
        l,r = 0, len(nums)-1
        
        final = nums[0]
        
        while l <= r:
            #这里的base case一定要考虑到
            if nums[l] < nums[r]:
                final = min(final,nums[l])
                break
            
            mid = (l+r)//2
            
            final = min(nums[mid],final)
            
            if nums[mid] >= nums[l]:
                l = mid+1
            
            else:
                r = mid -1
        
        return final
