time:o(n)
space:o(1)

class Solution:
    def minFlipsMonoIncr(self, s: str) -> int:
        
        ones = 0
        flip = 0
        
        for ele in s:
            if ele == "1":
                ones = ones+1
                flip = min(flip,ones)
            
            else:
                flip = flip +1
                flip = min(flip,ones)
        
        return flip
