class Solution:
    def subArrayRanges(self, nums: List[int]) -> int:
        
        result = 0
        for i in range(len(nums)):
            minv = float("inf")  ##注意这里的位置 通过 example 1跑一遍即可知道
            maxv = float("-inf")
            for j in range(i,len(nums)):
                minv = min(minv,nums[j])
                maxv = max(maxv, nums[j])
                result = result + (maxv-minv)
        
        return result
