```
time:o(n)
space:o(n)

class Solution:
    def kthFactor(self, n: int, k: int) -> int:
        
        result = []

        for i in range(1, n+1):
            if n%i == 0:
                result.append(i)
        
        #base case
        if n < k or len(result) < k:
            return -1
        
        return result[k-1]
        


```
