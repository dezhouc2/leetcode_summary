class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        
        pos,neg,result = 0,0,0
        
        for i in nums:
            if i == 0:
                neg = 0
                pos = 0
            
            elif i > 0:
                pos = pos +1
                if neg > 0:
                    neg = neg+1           
            else:
                pos,neg = neg,pos
                neg = neg +1
                if pos > 0:
                    pos = pos +1
            
            result = max(result,pos)
        
        return result
