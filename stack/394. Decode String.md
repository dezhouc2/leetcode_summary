time:o(n)
space:o(n)


class Solution:
    def decodeString(self, s: str) -> str:
        
        #component consists of string + number
        #each of the element can only be "[","]",char, number
        
        result = []
        string = ""
        number = 0
        
        
        for ele in s:
            if ele == "[":
                result.append([number,string])
                string = ""
                number = 0
            
            elif ele == "]":
                num, ch = result.pop()
                string = ch + num * string
            
            elif ele >= "0" and ele <= "9":
                number = number * 10 + int(ele)
            
            else:
                string = string + ele
        
        return string
