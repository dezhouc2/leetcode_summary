class MaxStack:

    def __init__(self):
        self.maximum = SortedList()
        self.stack = SortedList()
        self.id = 0
        
    def push(self, x: int) -> None:
        self.maximum.add((x,self.id))
        self.stack.add((self.id,x))
        self.id = self.id+1

    def pop(self) -> int:
        id,val = self.stack.pop()
        self.maximum.remove((val,id))
        return val
    
    def top(self) -> int:
        return self.stack[-1][1]

    def peekMax(self) -> int:
        return self.maximum[-1][0]

    def popMax(self) -> int:
        a,b = self.maximum.pop()
        self.stack.remove((b,a))
        return a
