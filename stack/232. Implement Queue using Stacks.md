class MyQueue:

    def __init__(self):
        self.tail = []
        self.head = []

    def push(self, x: int) -> None:
        while self.head:
            self.tail.append(self.head.pop())   
        
        self.tail.append(x)
        while self.tail:
            self.head.append(self.tail.pop())

    def pop(self) -> int:
        return self.head.pop()  

    def peek(self) -> int:
        return self.head[-1]      
        
    def empty(self) -> bool:
        if len(self.head) == 0:
            return True
        else:
