class Solution:
    def isValid(self, s: str) -> bool:
        dc = {"(":")","{":"}","[":"]"}
        stack = []
        
        for ele in s:
            if ele in dc:
                stack.append(ele)
            
            else:
                if not stack or dc[stack[-1]] != ele:
                    return False
                
                stack.pop()
        
        if stack:
            return False
        
        return True
   
time:o(n)
space:o(n)
