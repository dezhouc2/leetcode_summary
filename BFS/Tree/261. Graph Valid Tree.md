```python

time: o(n) n is the numbers of node
space: o(n)

错误思路: 刚开始是通过在bfs的过程中遍历遇到在set() 里面的点就输出false

问题: 图是否为 tree -> 根据图论要满足三个条件
1. 点的数量为 边数的数量+1
2. 每一对nodes,得有路径
3. 每一对nodes,有且只有一条路径 即没有cycle 



class Solution:
    def validTree(self, n: int, edges: List[List[int]]) -> bool:
        from collections import defaultdict        

        viz = set()
        dic_ = collections.defaultdict(list)
        queue = [0]
        
        for ele in edges:
            dic_[ele[0]].append(ele[1])
            dic_[ele[1]].append(ele[0])
        
        #print(dic_)
        
        #check 是否有且只有一条路径
        if len(edges) != n-1:
            return False
        
        #看看每一对nodes 是否有路径 然后用set记录node
        while queue:
            v = queue.pop(0)
            viz.add(v)
            
            for j in dic_[v]:
                if j not in viz:
                    queue.append(j)
                    viz.add(j)
        
        if len(viz) == n:
            return True
        else:
            return False
```
