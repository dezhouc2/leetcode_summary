
```python
time:o(n) numbers of cells in isConnected
time:o(n) # elements in queue + # elements in set()


class Solution:
    def findCircleNum(self, isConnected: List[List[int]]) -> int:
        
        n = len(isConnected)
        count = 0
        viz = set()
        
        
        def bfs(row):
            queue = []
            queue.append(row)
            viz.add(row)
            
            while queue:
                r = queue.pop(0)
                
                for j in range(n):
                    if isConnected[r][j] == 1 and j not in viz:
                        queue.append(j)
                        viz.add(j)
                    
                
        
        for i in range(n):
            if i not in viz:
                bfs(i)
                count = count+1
        
        return count
```
