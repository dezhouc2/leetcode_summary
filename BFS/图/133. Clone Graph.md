```python
time: O(N+M) N is number of nodes and M is number of edges
space: O(N) number of nodes

思路:
利用hashmap 的key-value特性 - key为每一个图的节点, value为每一个图的对应的clone node

"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
""'

class Solution:
    def cloneGraph(self, node: 'Node') -> 'Node':
        
        if not node:
            return node
        
        queue = [] 
        queue.append(node)
        
        visit = dict()
        
        #这里是在dictionary里面clone这个节点
        visit[node] = Node(node.val, [])  //这里要注意这个clone的形式
        
        while queue:
            vertex = queue.pop(0)
            
            for element in vertex.neighbors:
                if element not in visit:
                    queue.append(element)
                    visit[element] = Node(element.val, [])
                
                #把clone的node的邻居节点给加上
                visit[vertex].neighbors.append(visit[element]) //注意这里加上的是clone的node所以是visit[element]
        
        
        return visit[node]



```
