如果有关图的问题基本上都是这样时间和空间复杂度
time:o(E+v) E = numbers of edge, V = numbers of verticles
space:o(E+V)

思路:
对每一个点遍历 -> 看这个点是否直接/间接连接到其他点(这里得建立图) -> 如果是的话继续遍历另外一个点直到这整个连通分量结束
几个连通分量就是输出几

class Solution:
    def countComponents(self, n: int, edges: List[List[int]]) -> int:
        
        # [0,1]
        # [1,2]
        # [3,4]
        # numbers of cc -> numbers of bfs execute
        # 从谁开始?
        # 对于每一个开始的点开始 这个点的直接和间接联系 加入viz 依此类推
        # q 对于图问题什么时候考虑建立图 什么时候不考虑建立图
        
        count = 0
        queue = []
        viz = set()
        
        dic_ = collections.defaultdict(list)
        
        #每一个点的边集
        for ele in edges:
            dic_[ele[0]].append([ele[1]])
            dic_[ele[1]].append([ele[0]])
        
        for i in range(n):
            if i not in viz:
                queue.append(i)
                count = count+1
                
                while queue:
                    v = queue.pop(0)
                    viz.add(v)
                    
                    for j in dic_[v]:
                        if j[0] not in viz:
                            viz.add(j[0])
                            queue.append(j[0])
        
        return count
