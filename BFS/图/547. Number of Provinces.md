time:o(n)
space:o(n)

卡住的点: 怎么才能让每一个点不看本身的点是否为1, 而遍历去看其他的点是否为1? -> 用hashset 记录即可

class Solution:
    def findCircleNum(self, isConnected: List[List[int]]) -> int:
        
        #计算连通分量的个数 -> 参考numbers of island的做法
        #进行一次bfs然后 增加count的个数
        
        queue = []
        viz = set()
        row = len(isConnected)
        count = 0
        
        def bfs(x):
            queue.append(x)
        
            while queue:
                i = queue.pop(0)

                for j in range(row):
                    if isConnected[i][j] == 1 and j not in viz:
                        queue.append(j)
                        viz.add(j)
        
        for x in range(row):
            if x not in viz:
                bfs(x)
                count = count+1          
        
        return count
