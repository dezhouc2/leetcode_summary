
```python
time: o(M^2*N) M is the length of each word and N is the numbers of words in the wordlist
space: o(M^2*N)


class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        
        #bfs
        if endWord not in wordList:
            return 0
        
        queue = [(beginWord,1)]
        
        hashset = set(wordList)
        
        #if beginWord in hashset:    //这里理论上来说是要加上的否则 count会加1
        #    hashset.remove(beginWord)
        
        while queue:
            word, count = queue.pop(0)
            if word == endWord:
                return count
            
            for i in range(len(word)):
                for j in range(26):
                    new = word[:i] + chr(ord("a")+j) + word[i+1:]   //这里注意把unicode转化为字母的技巧
                    
                    if new in hashset:
                        hashset.remove(new)
                        queue.append((new,count+1))

        
        return 0
```
