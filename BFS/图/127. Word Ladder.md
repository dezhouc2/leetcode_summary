
```python
time: o(n*c^2) n = len of wordlist; c = len of word
space:o(n*c^2)

--* 注意:
1.每一个单词中每一个字母的26个字母转换

2.遇到在viz的单词要remove 避免重复

class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        #find a shorest path -> using bfs
        
        #for the base case
        if endWord not in wordList:
            return 0
        
        queue = [(beginWord,1)]
        viz = set(wordList)
        
        while queue:
            word,count = queue.pop(0)
            if word == endWord:
                return count
            
            #iterate
            for i in range(len(word)):
                for j in range(26):
                    word_ = word[:i] + chr(ord("a")+j) + word[i+1:]
                    
                    if word_ in viz:
                        queue.append((word_, count+1))
                        viz.remove(word_)
            
        
        return 0
```
