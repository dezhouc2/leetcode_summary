time:o(k) numbers of elements in queue
space:o(n) numbers of elements in queue + numbers of elements in hashset

最短路径问题
注意: 这道题很容易超时 要注意一下语句和变量

class Solution:
    def racecar(self, target: int) -> int:
        
        queue = deque()
        queue.append((0,1,0)) #(position, speed, steps)
        seen = {(1,0)} #(position, speed)
        while queue:
            position, speed, steps = queue.popleft()
            if position == target:
                return steps
            
            # Accelerate
            new_position = position + speed
            new_speed = speed * 2
            new_state = (new_position, new_speed)
            if new_state not in seen:
                queue.append((new_state[0], new_state[1], steps + 1))
                seen.add(new_state)
                
            # Reverse
            if speed > 0:
                new_speed = -1
            else:
                new_speed = 1
            new_state = (position, new_speed)
            if new_state not in seen:
                queue.append((new_state[0], new_state[1], steps + 1))
                seen.add(new_state)
