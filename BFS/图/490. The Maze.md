time:o(m*n) m = numbers of row, n = numbers of col
space:o(m*n)

class Solution:
    def hasPath(self, maze: List[List[int]], start: List[int], destination: List[int]) -> bool:
        
        #if there exists a path from start point to destination
        # bfs/dfs
        # base case -> destination 不是固定卡住
        
        
        queue = []
        queue.append(start)
        vis1 = set()
        vis1.add((start[0], start[1]))
        row = len(maze)
        col = len(maze[0])

        while queue:

            r,c = queue.pop(0)
            if [r,c] == destination:
                return True

            for x,y in [(1,0), (-1,0), (0,1), (0,-1)]:
                a = r
                b = c
                while 0 <= a+x < row and 0 <= b+y < col and maze[a+x][b+y] == 0:
                    a = a+x
                    b = b+y
                
                if (a,b) not in vis1:
                    vis1.add((a,b))
                    queue.append((a,b))
 
        return False
