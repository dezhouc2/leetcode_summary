## 总结:
注意: base case 当n等于0,等于1的分别情况

class Solution:
    def fib(self, n: int) -> int:       
        
        if n == 0:
            return 0
        if n == 1:
            return 1
        
        dp = [0]*(n+1)
        
        dp[1] = 1
        for i in range(2,len(dp)):
            dp[i] = dp[i-1] + dp[i-2]
        
        return max(dp)
