思路:
考虑有什么知识点是从数值转换到 字母的 -> chr(number)
通过example1 -> 知道 chr(columnNumber-1 % 26 + ord("A")) 可以得到 "A"

通过example2 -> chr(columnNumber-1 % 26 + ord("A")) 可以得到"B" 然后再通过 columnNumber // 26 经历一个循环得到 "A"

time: o(1)??, space:o(n)

class Solution:
    def convertToTitle(self, columnNumber: int) -> str:
        
        # (cnum - 1) + ord("A") = "A"
        # 28%26 == 2 -> convert to B
        # 28//26 =1 -> convert to A
        result = []
        
        while columnNumber > 0:
            columnNumber = columnNumber - 1
            val = chr(columnNumber % 26 + ord("A"))
            result.append(val)
            columnNumber = columnNumber //26
        
        
        result = result[::-1]
        return "".join(result)
