镜像问题(一模一样的做法): 509. Fibonacci Number


方法一: dp
class Solution:
    def climbStairs(self, n: int) -> int:
        
        #dp[i] represents the dictinct numbers of ways to climb to top
        #its equal to total numbers of climbing 1 steps + total numbers of climbing 2 steps
        
        dp = [0]*(n+1)
        
        if n == 1:
            return 1
        dp[0] = 1
        dp[1] = 1
        
        for i in range(2,n+1):
            dp[i] = dp[i-1] + dp[i-2]

方法二: 用变量来代替空间
time:o(n) space:o(1)

class Solution:
    def climbStairs(self, n: int) -> int:
        
        #base case
        if n == 1:
            return 1
        
        x = 1
        y = 1
        result = 0
        
        for i in range(2,n+1):
            result = x+y
            x = y
            y = result
        
        return result
