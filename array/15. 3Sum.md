time:o(n^2)
space:o(1)

思路:
先nums.sort() 因为考虑用双指针去重 否则得用hashset去重

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        
        result = []
        nums.sort()
        
        for i in range(len(nums)):
            
            if i != 0 and nums[i] == nums[i-1]:
                continue
            l,r = i+1, len(nums)-1
            while l < r:
                val = nums[r] + nums[l] + nums[i]
                if val == 0:
                    result.append([nums[r],nums[l],nums[i]])
                    l = l+1
                    while nums[l] == nums[l-1] and l <r:
                        l = l+1

                elif val < 0:
                    l = l +1

                else:
                    r = r -1
        
        return result
