思路:
通过example1和2 知道考虑快指针对应的数值为2和0时分别和 k指针对应的数值进行调换

while fast <= k -> 当时考虑的是fast <= len(nums) ## 因为k刚好设置为nums最后一个数所以考虑 fast <= k （通过example1考虑到把0和2调换）


class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        
        #constant extra space -> two pointers
        # [0,0,2,1,1,2] -> [0,0, 1,1,2,2]
        # s  f
        
        
        fast,slow = 0,0
        k = len(nums)-1
        
        while fast <= k:
            
            if nums[fast] == 2:
                nums[fast],nums[k] = nums[k],nums[fast]
                fast = fast+1
                k = k-1
            
            elif nums[fast] == 0:
                nums[fast], nums[slow] = nums[slow],nums[fast]
                fast = fast+1
                slow = slow+1
            
            else:(这里当时没有考虑到, 也是通过example1可以考虑到当fast > k的时候停止循环)
                fast = fast+1
