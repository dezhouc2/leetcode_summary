注意:
#第一次做的时候不知道遍历谁和 不知道函数的输入
backtrack 函数中的输入: string,index -> string的话是因为输出的result里面有string; index是用于遍历到具体的元素


class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        result = []
        
        if digits == "":
            return result
        
        dic_ = {"2":"abc", "3":"def", "4":"ghi", "5":"jkl", "6":"mno", "7":"pqrs", "8":"tuv",
               "9":"wxyz"}
        
        
        def backtrack(string,index):
            if len(digits) == index:
                result.append(string[:])
                return
            
            for ele in dic_[digits[index]]:
                
                backtrack(string+ele,index+1)
                    
        
        
        backtrack("",0)
        return result
