
```python
time: o(n*2^n)
space: o(n)


注意:
这道题没有常见的递归结束, 只是通过for loop中循环的起点来控制递归是否结束

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        
        result = []
        curr = []
        
        def backtrack(cur,index):
            result.append(cur[:])
            
            #注意这里之前写的是 index+1
            #通过画图可以知道, 递归是基于for loop中的当前指针+1
            for i in range(index,len(nums)):
                cur.append(nums[i])
                
                backtrack(cur,i+1)
                cur.pop()
            

        
        backtrack(curr,0)
        return result
```
