
```python
time: 每一个数选择为 (n,k), 有k个数
space: o(n)


class Solution:
    def combine(self, n: int, k: int) -> List[List[int]]:
        
        #poss -> backtrack
        result = []
        curr = []
        
        def backtrack(cur,index):
            if len(cur) == k:
                result.append(cur[:])
                return
            
            for i in range(index,n+1):
                cur.append(i)
                backtrack(cur,i+1)
                cur.pop()
            
        
        backtrack(curr,1)
        return result
```
