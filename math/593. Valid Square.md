class Solution:
    def validSquare(self, p1: List[int], p2: List[int], p3: List[int], p4: List[int]) -> bool:
        
        if p1 == p2 or p1 == p3 or p1 == p4 or p2 == p3 or p2 == p4 or p3 == p4:
            return False
        
        edge_1 = (p2[0]-p1[0])**2 + (p2[1]-p1[1])**2
        edge_2 = (p3[0]-p1[0])**2 + (p3[1]-p1[1])**2
        edge_3 = (p4[0]-p1[0])**2 + (p4[1]-p1[1])**2
        edge_4 = (p2[0]-p3[0])**2 + (p2[1]-p3[1])**2
        edge_5 = (p2[0]-p4[0])**2 + (p2[1]-p4[1])**2
        edge_6 = (p3[0]-p4[0])**2 + (p3[1]-p4[1])**2
        
        result = [edge_1,edge_2,edge_3,edge_4,edge_5,edge_6]
        
        count = Counter(result)
        
        if len(count) ==2:
            return True
        else:
            return False

思路:
1.先把每一个坐标用纸画出来
2.观察到,正方形只有两条为最长且一样, 有四条边最短也一样 -> 用counter计算是否有两种边长即可
