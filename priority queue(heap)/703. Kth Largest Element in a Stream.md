class KthLargest:

    def __init__(self, k: int, nums: List[int]):
        self.size = k
        self.hp = []
        for ele in nums:
            heapq.heappush(self.hp,ele)
        
        while len(self.hp) > self.size:
            heapq.heappop(self.hp)
            

    def add(self, val: int) -> int:
        heapq.heappush(self.hp,val)
        while len(self.hp) > self.size:
            heapq.heappop(self.hp)
        
        return self.hp[0]
