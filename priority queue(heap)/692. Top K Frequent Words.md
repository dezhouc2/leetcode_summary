time: o(k*logk)
space:o(n)

class Solution:
    def topKFrequent(self, words: List[str], k: int) -> List[str]:
        
        result = []
        
        hp = []
        count = Counter(words)
        
        for ele in set(words):
            heapq.heappush(hp, (-count[ele], ele))
        
        while len(result) < k:
            val = heapq.heappop(hp)[1]
            result.append(val)
        
        
        return result
