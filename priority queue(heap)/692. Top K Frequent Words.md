class Solution:
    def topKFrequent(self, words: List[str], k: int) -> List[str]:
        
        result = []
        hp = []
        
        dicty = collections.defaultdict(int)
        for ele in words:
            dicty[ele] = dicty[ele]+1
        
        for ele in dicty:
            heapq.heappush(hp, (-dicty[ele],ele))
        
        while len(result) < k:
            word = heapq.heappop(hp)
            result.append(word[1])
        
        #print(result)
        return result
