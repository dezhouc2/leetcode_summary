class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:
        
        def val(ele):
            return sqrt(ele[0]**2 + ele[1]**2)
        
        result = []
        hp = []
        
        for ele in points:
            heapq.heappush(hp, (val(ele),ele))
        
        while len(result) < k:
            val = heapq.heappop(hp)[1]
            result.append(val)
        
        return result
