class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:

        def euc(a,b):
            return sqrt(a*a+b*b)
        
        result = []
        hp = []
        
        for i in points:
            heapq.heappush(hp,(euc(i[0],i[1]),i))
        
        while len(result) < k:
            num = heapq.heappop(hp)
            result.append(num[1])
 
        return result
   
   
  
  note: 这道题的tuple不用dictionary的形式来寄存因为 会遇到有相同数值的坐标作为key 这样会有bug
