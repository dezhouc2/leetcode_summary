time:o(nlogn)
space:o(n)

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        
        result = []
        hp = []
        for ele in nums:
            heapq.heappush(hp, -ele)
        
        while hp:
            val = -heapq.heappop(hp)
            result.append(val)
        
        return result[k-1]
