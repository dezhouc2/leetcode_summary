
```python
time: o(nlogk) since o(n) requires to iterate the nums, and o(logk) for while k
space:o(n)

思路:
先转化成tuple的形式因为 heap需要 -> (count,val) 这里的count需要用counter来算出频率

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        
        hp = []
        count = Counter(nums)
        result = [(count[x],x) for x in set(nums)]
        #for ele in dic:
        #    heapq.heappush(hp,(-ele[1],ele[0]))
        
        for ele in result:
            heapq.heappush(hp,(-ele[0],ele[1]))
        
        output = []
        
        while len(output) < k:
            (c,val) = heapq.heappop(hp)
            output.append(val)
        
        return output
```
